using UnityEngine; using System.Collections.Generic; public class RunAndGunGame : MonoBehaviour { // --- Player --- public float moveSpeed = 8f; public float jumpForce = 12f; public Transform firePoint; public GameObject defaultBulletPrefab; private Rigidbody2D rb; private Animator animator; private bool isGrounded; // --- Weapons --- [System.Serializable] public class Weapon { public string name; public GameObject prefab; public int damage; public float fireRate; } public List weapons = new List(); private int currentWeaponIndex = 0; private float nextFireTime = 0f; // --- Player Stats --- public int health = 100; public int score = 0; public int level = 1; // --- Enemy --- public class Enemy : MonoBehaviour { public int health = 50; public GameObject deathEffect; public void TakeDamage(int damage) { health -= damage; if (health <= 0) Die(); } void Die() { if (deathEffect != null) Instantiate(deathEffect, transform.position, Quaternion.identity); Destroy(gameObject); } } void Start() { rb = GetComponent(); animator = GetComponent(); } void Update() { HandleMovement(); HandleJump(); HandleShoot(); HandleWeaponSwitch(); } void HandleMovement() { float move = Input.GetAxis("Horizontal"); rb.velocity = new Vector2(move * moveSpeed, rb.velocity.y); animator.SetFloat("Speed", Mathf.Abs(move)); } void HandleJump() { if (Input.GetButtonDown("Jump") && isGrounded) { rb.AddForce(Vector2.up * jumpForce, ForceMode2D.Impulse); animator.SetTrigger("Jump"); } } void HandleShoot() { if (Input.GetButton("Fire1") && Time.time >= nextFireTime) { Weapon currentWeapon = weapons.Count > 0 ? weapons[currentWeaponIndex] : null; if (currentWeapon != null) { Instantiate(currentWeapon.prefab, firePoint.position, firePoint.rotation); animator.SetTrigger("Shoot"); nextFireTime = Time.time + 1f / currentWeapon.fireRate; } else { Instantiate(defaultBulletPrefab, firePoint.position, firePoint.rotation); animator.SetTrigger("Shoot"); } } } void HandleWeaponSwitch() { if (Input.GetKeyDown(KeyCode.E)) { currentWeaponIndex = (currentWeaponIndex + 1) % weapons.Count; } } private void OnCollisionEnter2D(Collision2D collision) { if (collision.gameObject.CompareTag("Ground")) isGrounded = true; if (collision.gameObject.CompareTag("Enemy")) { health -= 10; if (health <= 0) Die(); } } private void OnCollisionExit2D(Collision2D collision) { if (collision.gameObject.CompareTag("Ground")) isGrounded = false; } void Die() { animator.SetTrigger("Die"); // Aquí puedes añadir lógica de reinicio de nivel o game over } // --- Progresión --- public void AddScore(int points) { score += points; if (score >= level * 1000) { LevelUp(); } } void LevelUp() { level++; health += 20; moveSpeed += 0.5f; // Aquí puedes desbloquear habilidades o armas nuevas } }
